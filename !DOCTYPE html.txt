<script>
/* =================================================================
   CONFIGURAÇÃO DO BANCO COMPARTILHADO (SUPABASE)
   Cole aqui a URL e a chave PUBLISHABLE/ANON do seu projeto.
   Pegue em: Supabase > seu projeto > Connect (ou Settings > API Keys)
   ================================================================= */
const SUPABASE_URL = "https://yojhpmsckyrmauswlgzm.supabase.co";   // ex: https://abcdxyz.supabase.co
const SUPABASE_KEY = "sb_publishable_ROVHCpHO-MLqG2vNzfObPA_jwPC4Ysp"; // chave publishable (sb_publishable_...) ou anon
/* ================================================================= */

(function(){
  const configurado = SUPABASE_URL.startsWith("http") && SUPABASE_KEY.length > 20;
  let db = null;
  if (configurado) {
    db = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);
  } else {
    document.getElementById('config-banner').style.display = 'block';
  }

  let registros = [];
  let chart = null;
  const isMobile = () => window.matchMedia('(max-width:560px)').matches;
  const fmt = n => 'R$ ' + Number(n).toLocaleString('pt-BR', {minimumFractionDigits:2, maximumFractionDigits:2});
  const today = () => new Date().toISOString().slice(0,10);
  const fmtData = d => { if(!d) return '—'; const [y,m,day]=d.split('-'); return `${day}/${m}/${y}`; };
  const MESES = ['jan','fev','mar','abr','mai','jun','jul','ago','set','out','nov','dez'];
  const mesLabel = ym => { const [y,m]=ym.split('-'); return `${MESES[+m-1]}/${y.slice(2)}`; };
  const esc = s => String(s).replace(/[&<>"]/g, c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;'}[c]));

  document.getElementById('in-data').value = today();

  async function load(){
    if(!db){ render(); return; }
    const { data, error } = await db.from('fretes').select('*').order('data', {ascending:false});
    if(error){ flash('Erro ao carregar do banco', true); registros=[]; }
    else { registros = data || []; }
    render();
  }

  function flash(text, err, id){
    const m = document.getElementById(id||'msg');
    m.textContent = text;
    m.style.color = err ? 'var(--danger)' : 'var(--success)';
    setTimeout(()=>{ m.textContent=''; }, 3000);
  }

  function totaisPorMes(){
    const map = {};
    registros.forEach(r=>{ if(r.data){ const ym=r.data.slice(0,7); map[ym]=(map[ym]||0)+Number(r.valor); } });
    return Object.entries(map).sort((a,b)=>a[0].localeCompare(b[0]));
  }
  function totaisPorLoja(){
    const map = {};
    registros.forEach(r=>{ map[r.loja]=(map[r.loja]||0)+Number(r.valor); });
    return Object.entries(map).sort((a,b)=>b[1]-a[1]);
  }

  function renderChart(){
    const dados = totaisPorMes();
    const ctx = document.getElementById('grafico');
    if(chart) chart.destroy();
    chart = new Chart(ctx, {
      type:'bar',
      data:{ labels:dados.map(d=>mesLabel(d[0])), datasets:[{ data:dados.map(d=>Math.round(d[1]*100)/100), backgroundColor:'#378ADD', borderRadius:6, maxBarThickness:48 }] },
      options:{ responsive:true, maintainAspectRatio:false,
        plugins:{ legend:{display:false}, tooltip:{ callbacks:{ label:c=>'R$ '+c.parsed.y.toLocaleString('pt-BR',{minimumFractionDigits:2}) } } },
        scales:{ x:{ grid:{display:false}, ticks:{ autoSkip:false, maxRotation:45, font:{size:isMobile()?10:12} } },
                 y:{ beginAtZero:true, ticks:{ callback:v=>'R$ '+v.toLocaleString('pt-BR'), font:{size:isMobile()?10:12} } } }
      }
    });
  }

  function render(){
    const total = registros.reduce((s,r)=>s+Number(r.valor),0);
    const lojasArr = totaisPorLoja();
    document.getElementById('stat-total').textContent = fmt(total);
    document.getElementById('stat-lojas').textContent = lojasArr.length;
    document.getElementById('stat-qtd').textContent = registros.length;

    const max = lojasArr.length ? lojasArr[0][1] : 1;
    const cont = document.getElementById('por-loja');
    cont.innerHTML = !lojasArr.length ? '<p class="empty">Nenhum frete registrado ainda.</p>' :
      lojasArr.map(([loja,val])=>{
        const pct = Math.round((val/max)*100);
        return `<div class="bar-row"><div class="bar-top"><span class="nm">${esc(loja)}</span><span class="vl">${fmt(val)}</span></div><div class="bar-track"><div class="bar-fill" style="width:${pct}%"></div></div></div>`;
      }).join('');

    const filtro = document.getElementById('in-filtro').value.toLowerCase();
    const hist = registros.filter(r=>r.loja.toLowerCase().includes(filtro))
      .slice().sort((a,b)=>(b.data||'').localeCompare(a.data||''));
    const mob = isMobile();
    const h = document.getElementById('historico');
    if(!hist.length){ h.innerHTML = '<p class="empty">Sem registros.</p>'; }
    else {
      h.innerHTML = `<table><thead><tr>
        <th style="width:${mob?'34%':'26%'}">Loja</th>
        <th style="width:${mob?'28%':'20%'}">Valor</th>
        <th style="width:${mob?'26%':'18%'}">Data</th>
        ${mob?'':'<th class="obs-col" style="width:28%">Obs.</th>'}
        <th style="width:${mob?'12%':'8%'}"></th>
      </tr></thead><tbody>
      ${hist.map(r=>`<tr>
        <td style="font-weight:500">${esc(r.loja)}</td>
        <td>${fmt(r.valor)}</td>
        <td style="color:var(--muted)">${fmtData(r.data)}</td>
        ${mob?'':`<td class="obs-col" style="color:var(--muted)">${esc(r.obs||'—')}</td>`}
        <td><button class="del" data-del="${r.id}" aria-label="Excluir">🗑</button></td>
      </tr>`).join('')}
      </tbody></table>`;
      h.querySelectorAll('[data-del]').forEach(b=>{
        b.onclick = async ()=>{
          const id = b.getAttribute('data-del');
          if(db){ await db.from('fretes').delete().eq('id', id); }
          registros = registros.filter(r=>String(r.id)!==String(id));
          render();
        };
      });
    }
    renderChart();
  }

  document.getElementById('btn-add').onclick = async ()=>{
    const loja = document.getElementById('in-loja').value.trim();
    const valor = parseFloat(document.getElementById('in-valor').value);
    const data = document.getElementById('in-data').value;
    const obs = document.getElementById('in-obs').value.trim();
    if(!loja){ flash('Informe a loja', true); return; }
    if(!valor || valor<=0){ flash('Informe um valor válido', true); return; }
    if(!db){ flash('Configure o Supabase primeiro', true); return; }
    const novo = { loja, valor, data, obs };
    const { data:ins, error } = await db.from('fretes').insert([novo]).select();
    if(error){ flash('Erro ao salvar', true); return; }
    registros.unshift(ins[0]);
    document.getElementById('in-loja').value='';
    document.getElementById('in-valor').value='';
    document.getElementById('in-obs').value='';
    document.getElementById('in-data').value = today();
    flash('Frete salvo');
    render();
  };
  document.getElementById('in-filtro').oninput = render;
  document.getElementById('in-valor').addEventListener('keydown', e=>{ if(e.key==='Enter') document.getElementById('btn-add').click(); });
  let rt; window.addEventListener('resize', ()=>{ clearTimeout(rt); rt=setTimeout(render,200); });

  document.getElementById('btn-excel').onclick = ()=>{
    if(!registros.length){ flash('Sem dados', true, 'msg2'); return; }
    const wb = XLSX.utils.book_new();
    const det = [['Loja','Valor (R$)','Data','Observação']];
    registros.slice().sort((a,b)=>(a.data||'').localeCompare(b.data||'')).forEach(r=>det.push([r.loja, Number(r.valor), r.data?fmtData(r.data):'', r.obs||'']));
    det.push([]); det.push(['TOTAL GERAL', registros.reduce((s,r)=>s+Number(r.valor),0)]);
    const ws1 = XLSX.utils.aoa_to_sheet(det); ws1['!cols']=[{wch:22},{wch:14},{wch:12},{wch:30}];
    XLSX.utils.book_append_sheet(wb, ws1, 'Fretes');
    const pl = [['Loja','Total (R$)','Nº']]; const cnt={}; registros.forEach(r=>cnt[r.loja]=(cnt[r.loja]||0)+1);
    totaisPorLoja().forEach(([l,v])=>pl.push([l,v,cnt[l]]));
    const ws2 = XLSX.utils.aoa_to_sheet(pl); ws2['!cols']=[{wch:22},{wch:14},{wch:8}];
    XLSX.utils.book_append_sheet(wb, ws2, 'Por loja');
    const pm = [['Mês','Total (R$)']]; totaisPorMes().forEach(([m,v])=>pm.push([mesLabel(m),v]));
    const ws3 = XLSX.utils.aoa_to_sheet(pm); ws3['!cols']=[{wch:12},{wch:14}];
    XLSX.utils.book_append_sheet(wb, ws3, 'Por mês');
    XLSX.writeFile(wb, `registrofacil_${today()}.xlsx`);
    flash('Excel gerado', false, 'msg2');
  };

  document.getElementById('btn-pdf').onclick = ()=>{
    if(!registros.length){ flash('Sem dados', true, 'msg2'); return; }
    const { jsPDF } = window.jspdf; const doc = new jsPDF();
    const total = registros.reduce((s,r)=>s+Number(r.valor),0);
    doc.setFontSize(16); doc.text('RegistroFácil — Relatório de Fretes', 14, 18);
    doc.setFontSize(10); doc.setTextColor(100); doc.text(`Gerado em ${fmtData(today())}`, 14, 25); doc.setTextColor(0);
    doc.text(`Total: ${fmt(total)}  •  Lojas: ${totaisPorLoja().length}  •  Fretes: ${registros.length}`, 14, 32);
    doc.autoTable({ startY:38, head:[['Loja','Total (R$)']], body:totaisPorLoja().map(([l,v])=>[l,fmt(v)]), headStyles:{fillColor:[55,138,221]}, styles:{fontSize:9} });
    let y = doc.lastAutoTable.finalY + 8;
    doc.autoTable({ startY:y, head:[['Mês','Total (R$)']], body:totaisPorMes().map(([m,v])=>[mesLabel(m),fmt(v)]), headStyles:{fillColor:[55,138,221]}, styles:{fontSize:9} });
    y = doc.lastAutoTable.finalY + 8;
    doc.autoTable({ startY:y, head:[['Loja','Valor','Data','Obs.']], body:registros.slice().sort((a,b)=>(a.data||'').localeCompare(b.data||'')).map(r=>[r.loja,fmt(r.valor),r.data?fmtData(r.data):'—',r.obs||'—']), headStyles:{fillColor:[55,138,221]}, styles:{fontSize:8} });
    doc.save(`registrofacil_${today()}.pdf`);
    flash('PDF gerado', false, 'msg2');
  };

  document.getElementById('btn-export').onclick = ()=>{
    const payload = { _tipo:'registrofacil_banco', _versao:1, exportadoEm:new Date().toISOString(), registros:registros.map(r=>({loja:r.loja,valor:Number(r.valor),data:r.data,obs:r.obs||''})) };
    const blob = new Blob([JSON.stringify(payload,null,2)], {type:'application/json'});
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a'); a.href=url; a.download=`registrofacil_banco_${today()}.json`; a.click();
    URL.revokeObjectURL(url);
    flash('Banco exportado', false, 'msg2');
  };

  document.getElementById('btn-import').onclick = ()=> document.getElementById('file-import').click();
  document.getElementById('file-import').onchange = (e)=>{
    const file = e.target.files[0]; if(!file) return;
    const reader = new FileReader();
    reader.onload = async ()=>{
      try {
        const data = JSON.parse(reader.result);
        const novos = Array.isArray(data)?data:data.registros;
        const validos = (novos||[]).filter(r=>r&&r.loja&&typeof r.valor==='number')
          .map(r=>({loja:r.loja, valor:r.valor, data:r.data||today(), obs:r.obs||''}));
        if(!validos.length){ flash('Arquivo sem fretes válidos', true, 'msg2'); return; }
        if(!db){ flash('Configure o Supabase primeiro', true, 'msg2'); return; }
        if(!confirm(`Importar ${validos.length} fretes para o banco compartilhado?`)) return;
        const { error } = await db.from('fretes').insert(validos);
        if(error){ flash('Erro ao importar', true, 'msg2'); return; }
        await load();
        flash(`${validos.length} fretes importados`, false, 'msg2');
      } catch(err){ flash('Arquivo inválido', true, 'msg2'); }
      e.target.value='';
    };
    reader.readAsText(file);
  };

  load();
})();
</script>
